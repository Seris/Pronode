 14 /**
 15  * Represent an Node.JS Application
 16  * @constructor
 17  * @param {string} name - Name of the application
 18  * @param {object} options - Options of the application
 19  */
 20 var Application = function(name, options){
 21     // Run all in a domain to catch all error
 22     this.domain = domain.create();
 23     this.domain.add(this);
 24     this.domain.run(function(){
 25         // Application name
 26         this.name = name;
 27         // Entry Point
 28         this.entry_point = options.entry_point;
 29         console.log(options.entry_point);
 30         // Working Folder
 31         this.cwd = options.cwd || path.dirname(this.entry_point);
 32         // Environment
 33         this.env = options.env || {};
 34 
 35         // Options
 36         this.options = options;
 37 
 38         // Log File
 39         this.logStream = {};
 40         this.logStream.stdout = fs.createWriteStream(this.options.stdout || Pronode.path('applications/' + this.name + '/stdout.out'), {flags: 'a'});
 41         this.logStream.stderr = fs.createWriteStream(this.options.stderr || Pronode.path('applications/' + this.name + '/stderr.out'), {flags: 'a'});
 42 
 43         this.started = false;
 44     }.bind(this));
 45 };

 52 /**
 53  * Start the wrapper of an application
 54  * @returns {function} Promise 
 55  */
 56 Application.prototype._spawnWrapper = function(){
 57     var self = this;
 58 
 59     // Start Wrapper process
 60     this.process = child_process.spawn(process.execPath, [Pronode.bin('Application/Wrapper'), this.entry_point].concat(this.options.argv || []), {
 61         cwd: this.cwd,
 62         env: this.env,
 63         detached: false,
 64         stdio: ['ipc']
 65     });
 66 
 67     // Pipe log to file
 68     this.process.stdout.pipe(this.logStream.stdout);
 69     this.process.stderr.pipe(this.logStream.stderr);
 70 
 71     // When the process exit
 72     this.process.once('exit', function(){
 73         self.started = false;
 74         this.emit('stop', null, code, signal);
 75         Pronode.ev.emit('application exit', CurrentApp);
 76         self.domain.dispose();
 77     });
 78 
 79     // When the domain dispose
 80     this.domain.on('dispose', function(){
 81         self.removeAllListeners();
 82     });
 83 
 84     this.started = true;
 85     this.aborted = false;
 86 
 87     return new Promise(function(resolve, reject){
 88         this.process.on('message', function(message){
 89             switch(message){
 90                 case 'wrapper-started':
 91                     this.removeListener('error', onError);
 92                     this.removeListener('exit', onExit);
 93                     resolve();
 94                     break;
 95             }
 96         });
 97 
 98         // On Error
 99         var onError = function(err){
100             this.removeListener('exit', onExit);
101             reject(err);
102         };
103         this.process.once('error', onError);
104 
105         // On Exit
106         var onExit = function(){
107             reject(new Error('Wrapper has exited unexpectedly !'));
108         };
109         self.process.once('exit', onExit);
110     }.bind(this));
111 };

125             self.process.send({

128             });
129 
130             self.started = true;
131             self.emit('start', null, this);
132         };
133 
134         var onReject = function(err){
135             self.emit('start', err);
136             self.abort(err);
137         };
138 
139         this._spawnWrapper().done(onResolve, onReject);
140     }.bind(this));

144 /**
145  * Stop the application
146  */
147 Application.prototype.stop = function(){
148     if(!this.started){
149         return;
150     }
151 
152     this.domain.run(function(){
153         // Force application to exit 10 seconds after the signal
154         var forceExit = setTimeout(function(){
155             this.process.kill('SIGKILL');
156         }, 10000);
157 
158         // Executed when the application exit
159         this.process.once('exit', function(code, signal){
160             clearTimeout(forceExit);
161         });
162 
163         // Ask to the application to stop nicely
164         this.process.kill('SIGHUP');   
165     }.bind(this));
166 };

168 /**
169  * Abort the application (useful on error)
170  */
171 Application.prototype.abort = function(err){
172     if(this.aborted){
173         return;
174     }
175 
176     this.aborted = true;
177     if(this.started){
178         // Try to kill the application
179         this.process.kill('SIGKILL');
180     }
181 
182     this.emit('abort', arguments[0]);
183 
184     // Dispose Domain
185     this.domain.dispose();
